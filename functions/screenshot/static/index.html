Options

  • width: number
    • default: 1920
  • height: number
    • default: 1080
  • scale: number
    • default: 1
  • darkMode: boolean
    • default: false
  • animation: string
    • default: 1000,500,0,smooth

    • animation format:

      • wait: number
      • top: number
      • left: number
      • behavior: auto | instant | smooth
    • example: animation=1000,100,0,smooth (wait 1000 ms, scroll down from top 100 units, scroll from left 0 units, scroll behavior smooth)
